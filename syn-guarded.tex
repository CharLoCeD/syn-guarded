\section{Introduction}

\subsection{Axiomatisation}

This is an internal analysis on synthetic guarded domain theory~\cite{birkedal2012first}. Let $\W$ be the theory of filters on the poset $\omega = (\N,\le) = \set{0 < 1 < \cdots}$. For convenience, we also introduce the following related poests, 
\[ \ovob = \set{\bot < 0 < 1 < \cdots}, \quad \omega^\infty = \set{0<1<\cdots<\infty}. \]
Explicitly, $\W$ has countably many propositional constants $\ms p_\bot,\ms p_0,\ms p_1,\cdots$, with axioms:
\begin{enumerate}
  \item[(w1)] $\ms p_\bot \vdash \bot$;
  \item[(w2)] For all $n\le m\in\ovob$, $\ms p_n \vdash \ms p_m$;
  \item[(w3)] $\top \vdash \bigvee_{n\in\omega} \ms p_n$.
\end{enumerate}
We work internally in the classifying topos $\Set[\W] \simeq \psh(\omega)$. We will use $\Prop$ to denote the type of propositions. We will also assume an increasing sequence of univalent type universes $\Type$. We will use $\ovob$ to denote the poset internalised in $\Set[\W]$ as well. For any $n:\ovob$, we use $\ass{n}$ to denote the representable proposition.

\begin{axiom}[QC]\label{sgdqc}
  For any $n,m:\ovob$, 
  \[
  \ass n \to \ass m =
  \begin{cases}
    \ass m & m < n \\ 
    1 & \other
  \end{cases}
  \]
\end{axiom}

\begin{axiom}[GR]\label{sgdqcc}
  Every proposition is definable: $\fa p\Prop \ex n{\ovob} p = \ass{n}$.
\end{axiom}

\begin{axiom}[PG]\label{projective}
  For any $n:\omega$, $\ass n$ is projective.
\end{axiom}

\begin{remark}
  The three axioms above are of different nature: 
  \begin{itemize}
    \item (QC) is an ``algebraic'' axiom, in the sense that it is an axiom stable under quotients of theories. (QC) is in fact already valid in the classifying topos $\psh(\omega^\infty)$ of the Horn theory with the same signature with only axioms (w1) and (w2). In fact, (QC) will be valid in any subtopos containing the generic filter in $\psh(\omega^\infty)$.
    \item On the other hand, (GR) is characteristic of the classifying topos $\Set[\mbb W]$. From a logical perspective, it holds since geometric formulas in $\mbb W$ have very simple normal forms, i.e. it is either $\bot,\top$, or $\ms p_n$ for some $n\in\omega$. 
    \item (PG) holds since $\Set[\mbb W] \simeq \psh(\omega)$ is of presheaf type, where the representables are internally projective.
  \end{itemize}
\end{remark}

\begin{remark}
  Globally, we will assume the two axioms (QC) and (GR). When we need to invoke the projectivity axiom (PG), we will label it explicitly.
\end{remark}

\subsection{Elementary properties}

In fact, these two axioms implies that the family of propositions form a model of $\W$ internally, i.e. they form a filter of $\omega$.

\begin{proposition}\label{ismodel}
  The family of propositions $\set{\ass n}_{n:\ovob}$ is a model of $\W$.
\end{proposition}
\begin{proof}
  By (GR), $\emp = \ass n$ for some $n \colon \ovob$, and by (QC) $\ass\bot \to \ass n = 1$, which implies $\ass\bot \to \emp$, thus $\ass\bot = \emp$. This shows (w1) holds. (w2) is a direct consequence of (QC). For (w3), by (GR) again, $\ex n{\ovob} 1 = \ass n$, while $1 = \ass n$ is equivalent to $\ass n$, thus this implies $\bigvee_{n:\ovob}\ass n$ holds.
\end{proof}

In fact, this family of propositions should form a \emph{generic} filter. One evidence is the following simple observation:

\begin{lemma}\label{dnegclose}
  For any $n:\omega$, $\neg\neg\ass n$.
\end{lemma}
\begin{proof}
  By (QC), given $n:\omega$, $\ass n \to \ass\bot = \ass\bot$, which by~\cref{ismodel} is equivalent to $\emp$. Thus, $\neg\neg\ass n$.
\end{proof}

\begin{corollary}\label{wlem}
  The weak excluded middle holds:
  \[ \fa p{\Prop} \neg p \vee \neg\neg p. \]
\end{corollary}
\begin{proof}
  Since we are proving a proposition, for any $p:\Prop$ we can choose $n:\ovob$ that $p = \ass n$. Now if $n = \bot$ then $p = \ass\bot = \emp$, thus $\neg p$ holds; if $n \ge 0$, then by~\cref{dnegclose} $\neg\neg p$ holds. Thus, $\neg p \vee \neg\neg p$ holds for any $p:\Prop$.
\end{proof}

\begin{remark}\label{empdecidable}
  In other words,~\cref{wlem} shows that $\Prop$ can be written as a coproduct
  \[ \Prop \simeq \set{\emp} + \Prop_{\neg\neg}. \]
  This means for $p:\Omega$, $p = \emp$ is \emph{decidable}. In particular, for $n:\ovob$, $\ass n = \emp$ iff $n = \bot$.
\end{remark}

\begin{proposition}\label{doubleneg}
  $\neg\neg \colon \Prop \to \Prop$ is equal to $p \mapsto \ass0 \to p$. In particular, it is continuous.
\end{proposition}
\begin{proof}
  We first show that $\neg\neg p = \ass 0 \to p$, for any $p:\Prop$. By (GR) and funext, it suffices to show this for all $\ass n$ for $n:\ovob$. 
  \begin{itemize}
    \item When $n = \bot$, by~\cref{ismodel} $\ass\bot = \emp$ and $\neg\neg\ass\bot = \emp$. On the other hand, by (QC) $\ass0 \to \ass\bot=\ass\bot = \emp$. This shows they coincide.
    \item When $n\ge 0$, then by~\cref{dnegclose} $\neg\neg\ass n$ holds. By (QC) $\ass0 \to \ass n$ also holds, thus again they coincide.
  \end{itemize}
  Now since $\ass0 \to (-)$ is a right adjoint, it automatically preserves all meets.
\end{proof}

\begin{corollary}\label{Kuroda}
  The Kuroda principle holds: for any $p \colon X \to \Prop$, 
  \[ \fa xX \neg\neg p(x) \to \neg\neg \fa xX p(x). \]
\end{corollary}
\begin{proof}
  By~\cref{doubleneg}, $\neg\neg$ preserves all limits.
\end{proof}

\begin{corollary}\label{negnegBoo}
  The double negation of LEM holds: $\neg\neg\fa p{\Prop}p \vee \neg p$.
\end{corollary}
\begin{proof}
  By~\cref{Kuroda}, it suffices to show for any $p:\Prop$, $\neg\neg(p \vee \neg p)$, or equivalently, $\neg\neg p \vee \neg p$. This holds by~\cref{wlem}.
\end{proof}

\begin{remark}\label{scatterlogic}
  \cite{ESAKIA200097} introduced the notion of \emph{$\bot$-scattered} and \emph{scattered} topos, and shows that the statements in \cref{Kuroda} and~\cref{negnegBoo} are in fact equivalent, in which case the topos is called $\bot$-scattered. As a consequence of~\cref{canonicallob} and~\cref{canonicalneg} later, we can in fact show internally that we are working with a \emph{scattered} topos.
\end{remark}

\begin{lemma}\label{internallyconnected}
  For any $n:\omega$, $\ass n$ is connected, i.e. $2 \simeq 2^{\ass n}$. 
\end{lemma}
\begin{proof}
  Given any map $\ass n \to 2$, this is equivalent to two propositions $p,q$ that $p \to \ass n$ and $q \to \ass n$ and that $p \wedge q = \emp$. By (GR), we may assume $p = \ass i$ and $q = \ass j$. This way, $\ass i \wedge \ass j = \ass{i \wedge j} = \emp$, which by~\cref{empdecidable}, $i \wedge j = \bot$, equivalently $i = \bot$ or $j = \bot$. Thus, the map $\ass n \to 2$ factors through some $1 \inj 2$, thus $2 \simeq 2^{\ass n}$.
\end{proof}

In fact, we can show a stronger result:

\begin{lemma}\label{interallyconconnected}
  For any $n:\omega$, $\N \simeq \N^{\ass n}$.
\end{lemma}
\begin{proof}
  Given $f \colon \ass n \to \N$, again we can define a sequence of propositions $p \colon \N \to \Prop$, where for any $i:\N$,
  \[ p(i) \coloneq f\inv(\set{i}). \]
  By construction, $\bigvee_{i:\N}p(i) = \ass n$ and $p(i) \wedge p(j) = \emp$ for all $i \neq j$. The first condition implies that there exists $i:\N$ that $p(i) = \ass n$. The second condition implies that $p(j) = \emp$ for all $j \neq i$. By~\cref{empdecidable}, such $i$ is unique, thus $f$ factors through a unique $\set i \hook \N$.
\end{proof}


\section{Later modality on propositions}

Given the two axioms (QC) and (GR), we can construct \emph{internally} a later modality on propositions, starting from some endo-function on $\ovob$. Later on we will also generalise the construction to the level of general types. In fact, we will construct a \emph{dependent} modality. The modality we construct will always be (well-)pointed and left exact. We will also characterise when it also satisfies \emph{L\"ob induction}.

Interestingly, we will provide both a \emph{positive} and \emph{negative} construction of of the later modality, so that it has both a mapping-out and mapping-in universal property.

\subsection{A positive construction}

\begin{construction}[A positive construction of the later modality on propositions]\label{consprop}
  Let $\theta \colon \ovob \to \ovob$ be a \emph{montone, inflationary} function: 
  \begin{itemize}
    \item For any $n,m:\ovob$ with $n\le m$, $\theta n \le \theta m$;
    \item For any $n:\ovob$, $n \le \theta n$.
  \end{itemize}
  Then the modality $\bbx_\theta \colon \Prop \to \Prop$ takes $p:\Prop$ to
  \[ \bbx_\theta p \coloneq \bigvee_{n:\ovob} \ass{\theta n} \wedge (\ass n \to p). \]
\end{construction}

\begin{proposition}\label{modalityonrepre}
  $\bbx_\theta \colon \Prop \to \Prop$ makes the following diagram commute,
  \[ 
  \begin{tikzcd}
    \ovob \ar[d, "\ass-"'] \ar[r, "\theta"] & \ovob \ar[d, "\ass-"] \\
    \Prop \ar[r, "\bbx_\theta"'] & \Prop
  \end{tikzcd}
  \]
\end{proposition}
\begin{proof}
  For any $n:\ovob$, by~\cref{consprop} we have 
  \begin{align*}
    \bbx_\theta\ass n 
    &= \bigvee_{m:\ovob} \ass{\theta m} \wedge (\ass m \to \ass n) \\ 
    &= \bigvee_{m \le n} \ass{\theta m} \vee \bigvee_{m > n} \ass{\theta m} \wedge \ass n \\ 
    &= \ass{\theta n} \vee \ass n \\ 
    &= \ass{\theta n}
  \end{align*}
  The first equality is by definition; the second is by (QC); the third by (QC) and the fact that $\theta$ is monotone and inflationary; similarly for the last equality.
\end{proof}

\begin{remark}\label{usualmodalitypos}
  For the usual later modality $\bbx$ in synthetic guarded domain theory as specified in~\cite{birkedal2012first}, it corresponds to the monotone inflation map $n \mapsto n\plo$ on $\ovob$. In particular, it can be internally defined as  
  \[ \bbx p \coloneq \bigvee_{n:\ovob}\ass{n\plo} \wedge (\ass n \to p). \]
  \cref{lobprop} will provide an internal proof of L\"ob induction for this operator.
\end{remark}

As a start, we can show that the modality $\bbx_\theta$ will always satisfy some desired property:

\begin{proposition}\label{posproplex}
  For any monotone and inflationary map $\theta \colon \ovob \to \ovob$, $\bbx_\theta$ is pointed and left exact.
\end{proposition}
\begin{proof}
  To show $\fa p{\Prop}p \to \bbx_\theta p$, by (GR) it suffices to show $\ass n \to \bbx_\theta\ass n$ for all $n:\ovob$. By~\cref{modalityonrepre} and (QC) this holds since $\bbx_\theta\ass n = \ass{\theta n}$ and $\theta$ is inflationary. 
  
  For left exactness, by (GR) we may assume $1 = \ass n$, thus by~\cref{modalityonrepre} we have $\bbx_\theta 1 = \bbx_\theta\ass n = \ass{\theta n}$. Since $\theta$ is inflationary, by (QC) this means $\ass{\theta n}$ holds as well, thus $\bbx_\theta 1 = 1$. Proving $\bbx_\theta$ preserves binary meets is similar.
\end{proof}

Most importantly, we have a criterion of when $\bbx_\theta$ satisfies L\"ob induction:

\begin{theorem}\label{lobprop}
  If $\theta$ furthermore is strictly inflationary, i.e. $n < \theta n$ for all $n:\ovob$, then $\bbx_\theta$ satisfies L\"ob induction:
  \[ \fa p{\Prop} (\bbx_\theta p \to p) \to p. \]
\end{theorem}
\begin{proof}
  Take $p:\Prop$, and by (GR) let $p = \ass n$. Assume $\bbx_\theta p \to p$, viz. $\ass{\theta n} \to \ass{n}$. Since $\theta$ is strictly increasing, by (QC) this implication is equivalent to $\ass n$, viz. $p$, holds.
\end{proof}

On the other hand, we shall also provide a \emph{negative} construction of the later modality. For this we introduce the notion of \emph{flat} maps.

\begin{definition}[Flat map]\label{flat}
  We say a monotone function $\rho \colon \ovob \to \ovob$ is \emph{flat}, if it is unbounded, i.e. for any $n:\ovob$, there exists $m:\ovob$ that $n < \rho m$. 
\end{definition}

\begin{definition}\label{dualmapofflat}
  Given a flat map $\rho$, by the least number principle we have a dual map $\qsi\rho \colon \ovob \to \ovob$ that for any $n:\ovob$, $\qsi\rho n$ is the least $m$ that $n < \rho m$. We also define the \emph{normalised} dual map $\pf\rho$ as $\pf\rho(n) \coloneq \max(\qsi\rho(n),n)$.
\end{definition}

\begin{remark}
  By definition, $\pf\rho$ is monotone and inflationary.
\end{remark}

\begin{construction}[A negative construction of the later modality on propositions]\label{consnegprop}
  Let $\rho \colon \ovob \to \ovob$ be a \emph{flat} map.
  Then the modality $\bbx^\rho \colon \Prop \to \Prop$ takes $p:\Prop$ to
  \[ \bbx^\rho p \coloneq \bigwedge_{n:\ovob} \ass{n} \vee (\ass{\rho n} \to p). \]
\end{construction}

\begin{remark}
  The form of the construction of modality in~\cref{consnegprop} is evidently \emph{dual} to that in~\cref{consprop}.
\end{remark}

Again, let us compute the value of $\bbx^\rho$ on the generic propositions.

\begin{proposition}\label{neglateronrep}
  Let $\rho$ be a flat map and $\pf\rho$ its normalised dual map. For any $n:\ovob$,
  \[ \bbx^\rho\ass n = \ass{\pf\rho n}. \]
\end{proposition}
\begin{proof}
  By (QC),~\cref{consnegprop} and the definition of $\qsi\rho,\pf\rho$ in~\cref{dualmapofflat}, we compute
  \begin{align*}
    \bbx^\rho\ass n 
    &= \bigwedge_{m:\ovob} \ass m \vee (\ass{\rho m} \to \ass n) \\ 
    &= \bigwedge_{m \ge \qsi\rho n} \ass m \vee \ass n \\
    &= \ass{\qsi\rho n} \vee \ass n  \\
    &= \ass{\pf\rho n} \qedhere
  \end{align*}
  The first identity holds by definition. The second holds since when $m < \qsi\rho n$, by definition $\rho m \le n$ thus the expression evaluates to 1; and by (QC), when $\rho m > n$, $\ass{\rho m} \to \ass n$ is equivalent to $\ass n$. The last identity holds by definition of $\pf\rho$.
\end{proof}

\begin{corollary}\label{negispos}
  For any flat map $\rho$, $\bbx^\rho = \bbx_{\pf\rho}$.
\end{corollary}
\begin{proof}
  By function extensionality, it suffices to show $\bbx^\rho p = \bbx_{\pf\rho}p$ for all $p:\Prop$. Now by (GR) it suffices to show this for all $\ass n$. The remaining holds by~\cref{neglateronrep} and~\cref{modalityonrepre}.
\end{proof}

\begin{corollary}
  For any flat map $\rho$, $\bbx^\rho$ is pointed and left exact.
\end{corollary}

\begin{theorem}\label{canonicallob}
  If $\rho$ is deflationary, i.e. $\rho n \le n$ for all $n:\ovob$, then $\bbx^\rho$ satisfies L\"ob induction.
\end{theorem}
\begin{proof}
  If $\rho$ is deflationary, then by monotonicity $\pf\rho$ must be strictly inflationary. Thus the claim holds by~\cref{lobprop} and~\cref{negispos}.
\end{proof}

\begin{remark}\label{canonicalneg}
  By~\cref{usualmodalitypos} and~\cref{negispos}, we have an alternative construction of the usual later modality $\bbx$. Notice that the identity map $\id \colon \ovob\to\ovob$ is flat, and its normalised dual map is exactly given by $n \mapsto n\plo$. This way, we have an alternative characterisation of $\bbx$ as below,
  \[ \bbx p \coloneq \bigwedge_{n:\ovob} \ass n \vee (\ass n \to p). \]
  This nice thing about this formula is that, by (GR), we can avoid mentioning the generic propositions at all,
  \[ \bbx p = \bigwedge_{q\colon\Prop} q \vee (q \to p). \]
  This is indeed the formula obtained in~\cite{ESAKIA200097}.
\end{remark}


\section{An investigation of the modalities}


Recall that for any proposition $p:\Prop$, it generates both an open modality $\opm p X \coloneq X^P$ and closed modality $\clm p X \coloneq p * X$, where $p*X$ is the \emph{join} of $p$ and $X$ defined as the following pushout,
\[\begin{tikzcd}
  {p \times X} & X \\
  p & {p * X}
  \arrow[tail, from=1-1, to=1-2]
  \arrow[from=1-1, to=2-1]
  \arrow["\gamma", from=1-2, to=2-2]
  \arrow["\iota"', tail, from=2-1, to=2-2]
  \arrow["\lrcorner"{anchor=center, pos=0.125, rotate=180}, draw=none, from=2-2, to=1-1]
\end{tikzcd}\]
Both of them are lex modalities in the sense of~\cite{lmcs:3826}. For any $n:\ovob$, we will also denote the open and closed modalities corresponding to $\ass n$ as $\opm n$ and $\clm n$, respectively. In fact, both of them are idempotent monads. Notice that since by (QC), $\ass n \to \ass{m}$ for any $n \le m$ in $\ovob$, there are thus canonical natural transformations 
\[ \gamma \colon \opm{m} \nt \opm{n}, \quad \iota \colon \clm{n} \nt \clm{m}. \]
We also write the units of these modalities as 
\[ \eta^{\opm n} \colon \id \nt \opm n, \quad \eta^{\clm n} \colon \id \nt \clm n. \]
For simplicity, we might also just write $\eta^{\opmm}$ and $\eta^{\clmm}$. Notice that $\gamma$ and $\iota$ are in fact natural transformations of \emph{monads}, thus they also commute with the units $\eta^\opmm$ and $\eta^\clmm$.

\begin{remark}\label{opclpresprop}
  In particular, since both $\opm n$ and $\clm n$ are lex modalities, they preserve propositions. For any proposition $p$, by construction $\opm n p = \ass n \to p$ and $\clm n p = \ass n \vee p$.
\end{remark}

\begin{remark}\label{modaltypes}
  As characterised in~\cite{lmcs:3826}, a type $X$ is $\opm n$-modal iff $X \to X^{\ass n}$ is an equivalence; a type $X$ is $\clm n$-modal iff $\ass n$ implies $X$ is contractible. By~\cref{internallyconnected,interallyconconnected}, all finite sets and $\N$ are $\opm n$-modal for $n:\omega$.
\end{remark}

\begin{lemma}\label{preservemodal1}
  If $n \ge m$, then $\opm n$ preserves $\clm m$-modal types.
\end{lemma}
\begin{proof}
  Suppose $X$ is $\clm m$-modal. By (QC), if $\ass m$ holds, then $\ass n$ holds as well, thus $\ass m$ implies $\opm nX \simeq X^{\ass n} \simeq X \simeq 1$, thus $\opm nX$ is again $\clm m$-modal.
\end{proof}

\begin{lemma}[PG]\label{preservemodal2}
  If $n < m$, then $\clm n$ preserves $\opm m$-modal types.
\end{lemma}
\begin{proof}
  Suppose $X$ is $\opm m$-modal. Consider the type $\clm nX$. To show it is $\opm m$-modal, suppose we have a map $f \colon \ass m \to \clm nX$. Recall from construction that $\clm nX$ is a pushout, thus admits a surjection as below,
  \[
  \begin{tikzcd}
    & \ass n + X \ar[d, two heads] \\
    \ass m \ar[r, "f"'] \ar[ur, dashed, "\ov f"] & \clm nX
  \end{tikzcd}
  \]
  By (PG), there exists a lift $\ov f \colon \ass m \to \ass n + X$. By~\cref{internallyconnected}, we can do a case distinction:
  \begin{itemize}
    \item If $\ov f$ factors through $\ass n$, then $\ass m \to \ass n$. By (QC), $\ass m \to \ass n = \ass n$, thus $\ass n$ holds. This implies $\clm n = \id$, and the claim trivialises.
    \item If $\ov f$ factors through $X$, then by $X$ being $\opm m$-modal there is a unique point $x:X$ making the following diagram commute,
    \[
    \begin{tikzcd}
      \ass m \ar[r, "\ov f"] \ar[d, hook] & X \\
      1 \ar[ur, dashed, "x"']
    \end{tikzcd}
    \]
    This way, $\eta^{\clmm}x : \clm nX$ gives an extension of $f$ as well. To show this extension is unique, again we can do a case distinction to test elements in $\clm nX$ coming from $\ass n$ or $X$; the latter case follows from $X$ being $\opm m$-modal, and the former case follows from the fact that $\clm nX$ will be contractible. \qedhere
  \end{itemize}
\end{proof}

\begin{lemma}\label{stronglydisjoint}
  When $m \le n$, then $\opm m$ is strongly disjoint from $\clm n$ in the sense of~\cite{lmcs:3826}, i.e. every $\clm n$-modal type is $\opm m$-connected.
\end{lemma}
\begin{proof}
  Let $X$ be $\clm n$-modal. Since $m \le n$, by (QC) $\ass m \to X\simeq 1$. In particular, this means $\opm mX \simeq X^{\ass m}$ will be contractible, thus $X$ will be $\opm m$-connected.
\end{proof}

\begin{theorem}\label{modalitycommute}
  The following properties of the closed and open modalities hold:
  \[
  \opm n\clm m =
  \begin{cases}
    \opm{\bot} & n \le m \\
    \clm m\opm n & n > m
  \end{cases}
  \]
\end{theorem}
\begin{proof}
  Take $n,m:\ovob$. Suppose $n \le m$. By~\cref{stronglydisjoint} $\opm n$ is strongly disjoint from $\clm m$, thus the composite $\opm n\clm m$, as their meets, is the zero modality $\opm{\bot}$. On the other hand, if $n > m$, then by~\cref{preservemodal1,preservemodal2}, $\opm n$ preserves $\clm m$-modal types and $\clm m$ preserves $\opm n$-modal types. By~\cite{lmcs:3826}, this implies both the composites $\opm n\clm m$ and $\clm m\opm n$ computes the meet of $\opm n$ and $\clm m$. Thus, they commute.
\end{proof}

\begin{definition}\label{fracture}
  For any $n,m:\ovob$ and any type $X:\Type$, following~\cite{lmcs:3826} there is a canonical \emph{fracture square} as below,
  \[
  \begin{tikzcd}
    X \ar[d, "\eta^{\clmm}"'] \ar[r, "\eta^{\opmm}"] & \opm m X \ar[d, "\eta^{\clmm}"] \\
    \clm n X \ar[r, "\clm n\eta^{\opmm}"'] & \clm n\opm m X
  \end{tikzcd}
  \]
  The diagram commutes by naturality of $\clm n$, and we will denote the unique map as 
  \[ \sigma_{n,m} \colon X \to \clm n\opm mX. \]
\end{definition}

\begin{remark}
  By~\cref{stronglydisjoint} and~\cite{lmcs:3826}, when $m \le n$, the fracture square in~\cref{fracture} will be a pullback.
\end{remark}

\section{Negative construction of later modality for types}

Let us now fix a flat and deflationary map $\rho \colon \ovob \to \ovob$.

\begin{construction}[Negative later modality on types]\label{negmodtype}
  For any $X\colon\Type$, define $\bx^\rho X$ to be the following end,
  \[ \bx^\rho X \coloneq \int_{n:\ovob} \clm n \opm{\rho n} X. \]
  The end is constructed as a limit, where for any $n:\ovob$ there is a projection map
  \[ \pi_n \colon \bx^\rho X \to \clm n \opm{\rho n}X, \]
  such that the following diagram commutes for any $n:\ovob$,
  \[
  \begin{tikzcd}
    \bx^\rho X \ar[r, "\pi_n"] \ar[d, "\pi_{n+1}"'] & \clm n \opm{\rho n}X \ar[d, "\iota"] \\ 
    \clm{n+1}\opm{\rho(n+1)}X \ar[r, "\clm{n+1}\gamma"'] & \clm{n+1}\opm{\rho n}X
  \end{tikzcd}
  \]
  Explicitly, $\bx^\rho X$ can be identified as the following type,
  \[ \bx^\rho X \coloneq \sum\prth{f : \prod_{n:\ovob} \clm n \opm{\rho n}X} \times \prod_{n:\ovob} \iota_{\opm{\rho n}X}(f_n) = (\clm{n+1}\gamma_{X})(f_{n+1}). \]
\end{construction}

\begin{remark}
  Notice that since $\rho\bot = \bot$ by being deflationary, and that $\opm\bot$ is the zero modality taking everything to $1$, we can also omit the $\bot$ index in the end,
  \[ \bx^\rho X \simeq \int_{n:\omega}\clm n\opm{\rho n}X. \]
\end{remark}

\begin{remark}[Functoriality of $\bx^\rho$]\label{funcofmodality}
  Given a map $f \colon X \to Y$, the functoriality of $\bx^\rho$ follows from the functoriality of $\int_{n:\ovob}$ and $\clm n\opm{\rho n}$:
  \[ \bx^\rho f \coloneq \int_{n:\ovob}\clm n\opm{\rho n}f. \]
\end{remark}

Our first goal is to show that $\bx^\rho$ coincide with $\bbx^\rho$ when applied to propositions. For this we first show that $\bx^\rho$ is left exact.

\begin{proposition}
  For any $p:\Prop$, $\bx^\rho p = \bbx^\rho p$.
\end{proposition}
\begin{proof}
  By~\cref{opclpresprop}, $\clm n p$ is simply the disjunction $\ass n \vee p$, thus is again a proposition for any $n:\ovob$. This shows that the identity types in these types are contractible. This way, we may compute that 
  \begin{align*}
    \bx^\rho p 
    &\simeq \sum\prth{f : \prod_{n:\ovob} \clm n \opm{\rho n} p} \times \prod_{n:\ovob} \iota f_n = (\clm{n+1}\gamma)(f_{n+1}). \\
    &\simeq \prod_{n:\ovob}\ass n \vee (\ass{\rho n} \to p) \\
    &\simeq \bbx^\rho p \qedhere
  \end{align*}
\end{proof}

\begin{corollary}\label{typelob}
  $\bx^\rho$ satisfies L\"ob induction.
\end{corollary}

Another quick observation is that $\bx^\rho$ is left exact:

\begin{proposition}\label{modalitypresfinitelimits}
  $\bx^\rho$ preserves finite limits.
\end{proposition}
\begin{proof}
  This is simply due to the construction of $\bx^\rho$ as $\int_{n:\ovob}\clm n\opm{\rho n}$, where both the modalities $\clm n$ and $\opm{\rho n}$ are left exact, and the limit $\int_{n:\ovob}$ commutes with any limits.
\end{proof}

\begin{theorem}
  $\bx^\rho$ is well-pointed.
\end{theorem}
\begin{proof}
  To construct a point $\eta^{\bx^\rho} \colon X \to \bx^\rho X$ for any $X \colon \Type$, we need to provide a map $\eta^{\bx^\rho}_n \colon X \to \clm n \opm{\rho n}X$ for any $n:\ovob$, with a proof for any $n:\ovob$, $\iota\eta^{\bx^\rho}_n = (\clm{n+1}\gamma)\eta^\bx_{n+1}$. We may take $\eta^{\bx^\rho}_n$ to be $\sigma_{n,\rho n}$ as in~\cref{fracture}. The commutativity follows from the following diagram,
  \[\begin{tikzcd}
    X & {\opm{\rho(n+1)}X} & {\clm n\opm{\rho(n+1)}X} & {\clm{n+1}\opm{\rho(n+1)}X} \\
    & {\opm{\rho n}X} & {\clm n \opm{\rho n}X} & {\clm{n+1}\opm{\rho n}X}
    \arrow["{\eta^{\opmm}}", from=1-1, to=1-2]
    \arrow["{\eta^{\opmm}}"', from=1-1, to=2-2]
    \arrow["{\eta^{\clmm}}"', from=1-2, to=1-3]
    \arrow["{\eta^{\clmm}}", curve={height=-18pt}, from=1-2, to=1-4]
    \arrow["{\eta^{\opmm}}", from=1-2, to=2-2]
    \arrow["\iota"', from=1-3, to=1-4]
    \arrow["{\clm n\gamma}", from=1-3, to=2-3]
    \arrow["{\clm{n+1}\gamma}", from=1-4, to=2-4]
    \arrow["{\eta^{\clmm}}"', from=2-2, to=2-3]
    \arrow["\iota"', from=2-3, to=2-4]
  \end{tikzcd}\]
  The left triangle commutes by construction; the two squares commutes by naturality; the side triangle commutes by $\iota$ being a monad map. 
  
  For well-pointedness, by~\cref{funcofmodality} the map $\bx^\rho\eta^{\bx^\rho}_X \colon \bx^\rho X \to \bx^\rho\bx^\rho X$ is given by 
  \[ \bx^\rho\eta^{\bx^\rho}_X = \int_{n:\ovob}\clm n\opm{\rho n}\eta^{\bx^\rho}_X. \]
  In other words, at component $n:\ovob$, $(\bx^\rho\eta^{\bx^\rho}_X)_n$ is given by the following composite,
  \[ 
  \begin{tikzcd}
    \bx^\rho X \ar[r, "\pi_n"] & \clm n\opm{\rho n}X \ar[r, "\clm n\opm{\rho n}\eta^{\bx^\rho}_X"] \ar[dr, "\pair{\clm n\opm{\rho n}\sigma_{m,\rho m}}_{m}"'] & \clm n\opm{\rho n}\bx^\rho X \ar[d, "\simeq"] \\
    & & \int_{m}\clm n\opm{\rho n}\clm m\opm{\rho m}X
  \end{tikzcd}
  \]
  On the other hand, $\eta^{\bx^\rho}_{\bx^\rho X}$ at component $n$ is given by the composite
  \[
  \begin{tikzcd}
    \bx^\rho X \ar[r, "\sigma_{n,\rho n}"] \ar[d, "\simeq"'] & \clm n\opm{\rho n}\bx^\rho X \ar[d, "\simeq"] \\ 
    \int_{m}\clm m\opm{\rho m}X \ar[r, "\int_{m}\sigma_{n,\rho n}"'] & \int_{m}\clm n\opm{\rho n}\clm m\opm{\rho m}X
  \end{tikzcd}
  \]
  To show the two composite coincide, it suffices to show they coincide at each component $m:\ovob$, i.e. for any $n,m$ we want 
  \[
  \begin{tikzcd}
    \bx^\rho X \ar[d, "\pi_m"'] \ar[r, "\pi_n"] & \clm n\opm{\rho n}X \ar[d, "\clm n\opm{\rho n}\sigma_{m,\rho m}"] \\
    \clm m\opm{\rho m}X \ar[r, "\sigma_{n,\rho n}"'] & \clm n\opm{\rho n}\clm m\opm{\rho m}X
  \end{tikzcd}
  \]
  For this we distinguish two cases:
  \begin{itemize}
    \item If $\rho n \le m$, then by~\cref{modalitycommute} $\opm{\rho n}\clm m \simeq \opm{\bot}$, which means the target is contractible. In this case the commutativity is automatic.
    \item If $m < \rho n \le n$, then by~\cref{modalitycommute} again we have
    \[ \clm n\opm{\rho n}\clm m\opm{\rho m}X \simeq \clm n\clm m\opm{\rho n}\opm{\rho m}X \simeq \clm n\opm{\rho m}X. \]
    Now it is easy to see the diagram will be equal to 
    \[
    \begin{tikzcd}
      \bx^\rho X \ar[d, "\pi_m"'] \ar[r, "\pi_n"] & \clm n\opm{\rho n}X \ar[d, "\iota"] \\
      \clm m\opm{\rho m}X \ar[r, "\clm m\gamma"'] & \clm n\opm{\rho m}X
    \end{tikzcd}
    \]
    which commutes by the construction of the end $\bx^\rho$. \qedhere
  \end{itemize}
\end{proof}

We can also establish some other properties of the later modality $\bx^\rho$.

\begin{lemma}\label{preservetruncation}
  $\bx^\rho$ preserves $k$-truncated types.
\end{lemma}
\begin{proof}
  For (-2)- and (-1)-types this follows from left exactness. Now $\omega_\bot$ by construction is 0-truncated. By~\cite{lmcs:3826}, both $\clm n$ and $\opm n$ preserves $n$-truncated types, and $n$-truncated types are closed under dependent sums and dependent products.
\end{proof}

We can also compute the open modality applied to the later modality:

\begin{proposition}\label{openoflater}
  Let $n:\ovob$. Then for any type $X$, we have an equivalence
  \[ \opm n\bx^\rho X \simeq \int_{m\le n}\clm m\opm{\rho m}X. \]
\end{proposition}
\begin{proof}
  By construction, we have 
  \begin{align*}
    \opm n\bx^\rho X 
    &\simeq \opm n\int_{m:\omega} \clm m\opm{\rho m} X \\
    &\simeq \int_{m:\omega}\opm n\clm m\opm{\rho m} X \\ 
    &\simeq \int_{m<n}\clm m\opm{\rho m}X
  \end{align*}
  The first equivalence holds by construction; the second holds since $\opm n$ is a right adjoint, thus preserves arbitrary limits; the third equivalence holds by~\cref{modalitycommute}, where when $m \ge n$ we have $\opm n\clm m = \opm{\bot}$, thus does not contribute to the computation of the limit, and that $\opm n\opm{\rho m} = \opm{\rho m}$ since $\rho m \le m \le n$. 
\end{proof}

Given a flat and deflationary map $\rho$, let us use $\check\rho$ to denote an alternative map: $\check\rho n$ is the smallest $m$ that $n \le \rho m$. We can then compute the following:

\begin{theorem}
  If $X$ is $\opm n$-modal, then $\bx X$ is $\opm{\check\rho n\plo}$-modal. 
\end{theorem}
\begin{proof}
  Suppose $X$ is $\opm n$-modal. We show that $\bx^\rho X \simeq \int_{m\le\check\rho n}\clm m\opm{\rho m}X$. For this we note that for any $m > \check\rho n$, i.e. $\rho m \ge n$, since $X$ is $\opm n$-modal, it is also $\opm{\rho m}$-modal. This way, by construction the following diagram commutes,
  \[
  \begin{tikzcd}
    \bx^\rho X \ar[r, "\pi_{\pf\rho n}"] \ar[d, "\pi_{m}"'] & \clm{\check\rho n}\opm{\rho\check\rho n}X \simeq \clm{\check\rho n}X \ar[d, "\iota"] \\ 
    \clm{m}\opm{\rho m}X \simeq \clm mX \ar[r, "\simeq"'] & \clm{m}\opm{\rho\check\rho n}X \simeq \clm{m}X
  \end{tikzcd}
  \]
  Thus it follows that $\pi_m$ is completely determined by $\pi_{\pf\rho n}$. This implies that 
  \[ \bx^\rho X \simeq \int_{m\le\check\rho n}\clm m\opm{\rho m}X. \]
  By~\cref{openoflater}, this is also equivalent to $\opm{\check\rho n+1}\bx^\rho X$.
\end{proof}

\section{Positive construction of the later modality on types}

In this section we provide a positive construction of the later modality on general types as a constructor $\bx_\theta \colon \Type \to \Type$. We describe it as a higher inductive type, given the parameter $\theta \colon \ovob \to \ovob$ as a monotone, inflationary map on $\ovob$. For this purpose we also introduce another type constructor $\upm p$ for any $p:\Prop$. By construction it is the \emph{left adjoint} of $\opm p$, and is concretely given by $\upm pX \coloneq p \times X$ for any type $X$. As a left adjoint to a modality, $\upm p$ will be \emph{copointed}, which is given by the projection map. And again for $p \to q$, there is a natural transformation 
\[ \delta \colon \upm p \nt \upm q. \]

\begin{construction}[Positive construction of later modality on types]\label{constype}
  $\bx_\theta \colon \Type \to \Type$ is constructed as follows: For any $X : \Type$, 
  \[ \bx_\theta X \coloneq \int^{n:\ovob}\upm{\theta n} \opm{n} X. \]
  Explicitly, $\bx_\theta X$ is a coend, thus encoded by a higher inductive type. We can write down the higher inductive definition of $\bx_\theta X$: it has two constructors
  \begin{align*}
    &\nu : \prod_{n:\ovob} \upm{\theta n} \opm nX \to \bx_\theta X \\ 
    &\kappa : \prod_{n:\ovob} \prod_{x:\upm{\theta n}\opm{n+1}X} \nu_{n+1}(\delta x) = \nu_n(\upm{\theta n}\gamma x)
  \end{align*}
  These constructors are accompanied by the following elimination rules: Given
  \begin{itemize}
    \item $E : \bx_\theta X \to \Type$
    \item $e : \prod_{n:\ovob}\prod_{x:\upm{\theta n}\opm{n}X}E(\nu_nx)$
    \item $p : \prod_{n:\ovob}\prod_{x:\upm{\theta n}\opm{n+1}X}\kappa_n(x)\#_Ee_{n+1}(\delta x) = e_n(\upm{\theta n}\gamma x)$.
  \end{itemize}
  there will be a section $\ind_{E,e,p} : \prod_{x:\bx_\theta X}E(x)$ with the following computation rules: for any $n:\ovob$, $x:\upm{\theta n}\opm nX$ and $y:\upm{\theta n}\opm{n+1}X$,
  \begin{itemize}
    \item $\ind_{E,e,p} \nu_n(x) = e_n(x)$
    \item $\apd_{\ind_{E,e,p}}\kappa_n(y) = p_n(y)$.
  \end{itemize}
\end{construction}

\begin{remark}\label{recforcoend}
  We also have an accompanying recursion principle: Given the data
  \begin{itemize}
    \item $E : \Type$
    \item $e : \prod_{n:\ovob}\upm{\theta n}\opm{n}X \to E$
    \item $p : \prod_{n:\ovob}\prod_{x:\upm{\theta n}\opm{n+1}X}e_{n+1}(\delta x) = e_n(\upm{\theta n}\gamma x)$
  \end{itemize}
  there will be a map $\rec_{E,e,p} \colon \bx_\theta X \to E$ with the following computation rules: for any $n:\ovob$, $x:\upm{\theta n}\opm{n}X$ and $y:\upm{\theta n}\opm{n+1}X$,
  \begin{itemize}
    \item $\rec_{E,e,p}\nu_n(x) = e_n(x)$
    \item $\apd_{\rec_{E,e,p}}(\kappa_n(y)) = p_n(y)$.
  \end{itemize}
\end{remark}

Our first goal is to show that the type-theoretic modality given in~\cref{constype} restricts to the propositional modality given in~\cref{consprop}.

\begin{proposition}\label{twoconscoin}
  For any $p\colon\Omega$, $\bbx_\theta p = \bx_\theta p$.
\end{proposition}
\begin{proof}
  By (GR), it suffices to show this for $\ass n$. By construction and (QC), we have
  \begin{align*}
    \bx_\theta\ass n 
    &\equiv \int^{m:\ovob}\upm{\theta m}\opm m \ass n \\ 
    &= \bigvee_{m:\ovob}\ass{\theta m} \wedge (\ass m \to \ass n) \\
    &= \bigvee_{m\le n}\ass{\theta m} \vee \bigvee_{m > n} \ass{\theta m} \wedge \ass n \\
    &= \ass{\theta n}
  \end{align*}
  By~\cref{modalityonrepre}, this coincides with $\bbx_\theta\ass n$. 
\end{proof}

\begin{proposition}

\subsection{The dependent modality on propositions}

\begin{proposition}\label{openmodofprop}
  For any $p:\Prop$, $\opm p\Prop \simeq \Prop/p$, where under this equivalence we have
  \[
  \begin{tikzcd}
    \Prop \ar[d, "\eta^\opmm"'] \ar[dr, "-\wedge p"] \\ 
    \opm p\Prop \ar[r, "\simeq"'] & \Prop/p
  \end{tikzcd}
  \]
\end{proposition}
\begin{proof}
  By construction, for any $p:\Prop$ we have 
  \[ \opm p\Prop \simeq \Prop^p \simeq \mc P(p) \simeq \Prop/p. \]
  Under this equivalence, $\eta^\opmm$ becomes taking conjunction with $p$. 
\end{proof}

\begin{remark}
  Notice that~\cref{openmodofprop} holds generally, with nothing to do with our axiomatisation.
\end{remark}

By the positive construction of the modality, we have 
\[ \bx\Prop \simeq \int^{n:\omega}\upm{n+1}\opm n\Prop. \]
By construction, for any type $X$,
\[ (\bx\Omega \to X) \simeq \int_{n:\omega} \upm{n+1}\opm{n}\Prop \to X \simeq \int_{n:\omega} \opm n\Prop \to \opm{n+1}X. \]
In particular, we have
\[ (\bx\Prop \to \Prop) \simeq \int_{n:\omega}\opm n\Prop \to \opm{n+1}\Prop. \]
By~\cref{openmodofprop}, $\opm n\Prop \simeq \Prop/\ass n$ for all $n:\omega$, and the map $\bx \colon \Prop \to \Prop$ restricts to maps 
\[ \bx \colon \opm n\Prop \simeq \Prop/\ass n \to \Prop/\ass{n\plo} \simeq \opm{n+1}\Prop. \]
We denote the induced dependent modality as
\[ \later \colon \bx\Prop \to \Prop. \]
We first observe that it is the classifying map of the canonical point in $\bx\Prop$:

\begin{proposition}\label{laterclassify}
  The following square is a pullback 
  \[
  \begin{tikzcd}
    \bx 1 \ar[d, "\bx\top"'] \ar[r, "\simeq"] \ar[dr, pullback] & 1 \ar[d, "\top"] \\
    \bx\Prop \ar[r, "\later"'] & \Prop
  \end{tikzcd}
  \]
\end{proposition}
\begin{proof}
  Suppose we have $\star:\ass{n\plo}$ and $p:\Prop/\ass n$. If $\later_n(\star, p) = \bx p$ holds, then by (GR) let $p = \ass m$. It follows that $\ass{m\plo}$ holds, thus $\ass{m+1}$ holds. This way, by construction $\nu_n(\star,p) = \nu_m(\star,p) = \bx\top$.
\end{proof}

\begin{proposition}\label{lateremb}
  $\later \colon \bx\Prop \to \Prop$ is an embedding.
\end{proposition}
\begin{proof}
  We can relativise the argument in~\cref{laterclassify}.
\end{proof}

\begin{proposition}\label{computesucc}
  The following diagram commutes,
  \[
  \begin{tikzcd}
    \Prop \ar[dr, "\bx"] \ar[d, "\eta^\bx"'] \\
    \bx\Prop \ar[r, "\later"'] & \Prop
  \end{tikzcd}
  \]
\end{proposition}
\begin{proof}
  This follows by construction.
\end{proof}

\begin{theorem}
  $\later \colon \bx\Prop \to \Prop$ factors through $\Prop_{\neg\neg}$. In fact, it is an equivalence $\bx\Prop \simeq \Prop_{\neg\neg}$. In particular, $\Omega \simeq 1 + \bx\Omega$.
\end{theorem}
\begin{proof}
  Notice that the image of $\later$ are always of the form $\bx p$ for some $p$, and all $\bx p$ are $\neg\neg$-closed. By~\cref{computesucc}, $\later$ is surjective on $\Prop_{\neg\neg}$. By~\cref{lateremb} $\later$ is also an embedding, thus it is an equivalence onto $\Prop_{\neg\neg}$. The remaining follows from~\cref{wlem}.
\end{proof}
